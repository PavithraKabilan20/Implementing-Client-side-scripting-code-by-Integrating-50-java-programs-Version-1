Evaluate the Expression (a+b)*(c-d)

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class Main{
    public static void main(String[]args)throws IOException{
        
        BufferedReader reader=new BufferedReader(new InputStreamReader(System.in));
        System.out.print("Enter the value of a: ");
        String n1=reader.readLine();
        
        System.out.print("Enter the value of b: ");
        String n2=reader.readLine();
        
        System.out.print("Enter the value of c: ");
        String n3=reader.readLine();
        
        System.out.print("Enter the value of d: ");
        String n4=reader.readLine();

        int a=Integer.valueOf(n1);
        System.out.println("a: "+a);
        
        int b=Integer.valueOf(n2);
        System.out.println("b: "+b);
        
        int c=Integer.valueOf(n3);
        System.out.println("c: "+c);
        
        int d=Integer.valueOf(n4);
        System.out.println("d: "+d);
        
        int answer=(a+b)*(c-d);
        
        System.out.println("("+a+"+"+b+")*("+c+"-"+d+")="+answer);
}
}

Input:

Enter the value of a:4
Enter the value of b:3
Enter the value of c:5
Enter the value of d:2

Output:
(4+3)*(5-2)=21
