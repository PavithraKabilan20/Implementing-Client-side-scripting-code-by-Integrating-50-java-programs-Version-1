Convert the given meter to centimeter

import java.util.Scanner;

public class Main{
    public static void main(String[]args){
        
        Scanner sc=new Scanner(System.in);
        
        System.out.print("Enter the value of meter: ");
        int meter=sc.nextInt();
        
        int cm=meter*100;
        System.out.println("The converted value of " +meter+ "m is "+cm+"cm");
        
        
    }
}

Input:

Enter the value of meter: 8

Output:
The converted value of 8m is 800cm
